<? PHP
пространство имен Нигматулаев;

использовать core \ LogAbstract ;
использовать core \ LogInterface ;

Класс  MyLog  расширяет  LogAbstract,  реализует  LogInterface {

	публичная  статическая  функция  write () {
		return MyLog :: Instance () -> _write ();
	}

	публичный  статический  журнал функций  ( $ str ) {
		MyLog :: Instance () -> log [] = $ str ;
	}

	публичная  функция  _write () {
		echo  implode ( "\ n" ,; $ this -> log );
	}
}
